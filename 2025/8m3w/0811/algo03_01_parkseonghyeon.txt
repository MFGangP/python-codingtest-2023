# 오름차순 정렬을 구현해야된다.
# j의 범위는 정확하게 기억하고 있다.
# 버블정렬로 고쳐야하고 왜 고쳤는지 이유를 말해라
arr = [5, 1, 4, 2, 8]

n = len(arr)
for i in range(n):
    for j in range(i): # j의 범위는 수정할 필요 없음
        # arr[j]이 arr[j+1]보다 클 때 바꿔야하는데 작으면 자리를 바꾸고 있다.
        # 부등호를 반대로 써야함.
        # < 이렇게 쓰면 내림차순
        if arr[j] > arr[j+1]:
            # 일단 == 기호 잘 못 씀.
            # 대입을 해야하는데 비교 연산자를 쓰고있다.
            arr[j], arr[j+1] = arr[j+1], arr[j]
